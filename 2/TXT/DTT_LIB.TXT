CREATE OR REPLACE FUNCTION FN_TRG_HST_TRN_BOOKS()
 RETURNS trigger AS 
 $BODY$ 
 DECLARE
 vSEQ BIGINT; 
 BEGIN 
 SELECT COALESCE(MAX(HST_ID),0)+1 INTO vSEQ FROM HST_TRN_BOOKS ;
 IF TG_OP = 'INSERT' THEN 
 INSERT INTO HST_TRN_BOOKS SELECT vSEQ , NEW.* ; 
 ELSIF TG_OP = 'UPDATE' THEN 
 INSERT INTO HST_TRN_BOOKS SELECT vSEQ , NEW.* ; 

 END IF; 
 RETURN NEW; 
 END; 
 $BODY$ 
 LANGUAGE plpgsql ; 

 CREATE TRIGGER TRG_HST_TRN_BOOKS
 AFTER INSERT Or UPDATE 
 ON TRN_BOOKS
 FOR EACH ROW 
 EXECUTE PROCEDURE FN_TRG_HST_TRN_BOOKS() ;


ALTER TABLE TRN_BOOKS  DISABLE TRIGGER TRG_HST_TRN_BOOKS ;
